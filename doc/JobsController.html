def index
  filtered_jobs = JobFilter.new(params).filter_and_sort

  @jobs = filtered_jobs.eager_load(associated_tables)
                       .order(sort_order(params[:sort]))
                       .page(params[:page])
                       .per(20)

  @resources, @total_jobs = CategorySidebar.new(filtered_jobs, params).build
end
